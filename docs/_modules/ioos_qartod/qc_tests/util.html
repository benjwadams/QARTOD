[docs]def get_climatology(grouped_data, clim_table):
    """
    A factory function which returns a lookup table for climatologies
    based on grouped data.  Should take pandas groupby values for time keys
    and compare them to the climatology indices
    """
    vals = [(g, clim_table.get(grpVal)) for (grpVal, g) in grouped_data]
    for gr in vals:
        print(gr)
    chunks = [((g >= grpVal[0]) & (g <= grpVal[1])).astype('i4')
           if grpVal is not None else
           pd.Series(np.repeat(QCFlags.UNKNOWN, len(g)), g.index)
           for (g, grpVal) in vals]
    res = pd.concat(chunks)
    # replace 0s from boolean with suspect values
    res[res == 0] = QCFlags.SUSPECT
    return res
